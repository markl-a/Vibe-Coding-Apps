  const [todos, setTodos] = useState<Todo[]>([]);
  const [inputText, setInputText] = useState('');

  const addTodo = () => {

      setTodos([
        ...todos,
        { id: Date.now().toString(), text: inputText, completed: false },
      ]);

  };

  const deleteTodo = (id: string) => {

        </TouchableOpacity>
      </View>

      <FlatList
        data={todos}
        keyExtractor={item => item.id}
        renderItem={({ item }) => (
          <View style={styles.todoItem}>
